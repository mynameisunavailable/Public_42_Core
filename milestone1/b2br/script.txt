{
	# architecture and kernel version
	uname -a
	# number of physical processors
	echo "CPU physical: " $(grep "physical id" /proc/cpuinfo | sort -u | wc -l)
	# number of virtual processors
	echo "vCPU: $(grep -c "processor" /proc/cpuinfo)"
	#The current available RAM on your server and its utilization rate as a percentage.
	echo "Memory Usage: " $(free -m | grep "Mem:" | awk '{printf "%d/%dMB(%.2f%%)", $3, $2, $3/$2*100}')
	#•The current available storage on your server and its utilization rate as a percentage.
	df | awk '{used += $3; total += $2} END {used /= 1024*1024; total /= 1024*1024} END {printf\
	"Disk Usage: %.2f/%dGb(%.2f%%)\n", used, total, used/total*100}'
	#•The current utilization rate of your processors as a percentage.
	top -bn1 | grep "Cpu(s)" | tr ',' ' ' | awk '{printf "CPU load: %.2f%%\n", 100-$8}'
	#•The date and time of the last reboot.
	#who -b | awk '{printf "Last boot: %s %s\n", $3, $4}'
	echo Last boot: $(uptime -s)
	#•Whether LVM is active or not.
	lsblk | grep -q "lvm" && echo yes || echo no
	#•The number of active connections.
	echo Connections TCP: $(ss -ta | grep "ESTAB" | wc -l) ESTABLISHED
	#•The number of users using the server.
	w -h | wc -l
	#echo User log: $(users | tr ' ' '\n' | sort -u | wc -l)
	#•The IPv4 address of your server and its MAC (Media Access Control) address.
	echo Network: IP $(hostname -I | awk '{print $1}') $(ip a | grep "link/ether" | head -n 1 | awk '{printf "(%s)", $2}')
	#•The number of commands executed with the sudo program.
	echo Sudo: $(journalctl -t sudo -q | grep -c "COMMAND=") cmd
} | wall
