#include <unistd.h>

void    ft_putstr(char *str)
{
    while (*str)
    {
        write(1, &(*str), 1);
        str++;
    }    
}

int main (int argc, char **argv)
{


    if (argc < 2)
    {
        ft_putstr("File name missing.");
    }
    else if (argc > 2)
    {
        ft_putstr("Too many arguments.");
    }
    ft_putstr("Cannot read file.");
}
